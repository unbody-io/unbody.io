export const getStaticProps = async () => {
    const {data: {payload: introVideo}} = await unbody.get.videoFile
        .where({
            remoteId: "1X03E8Y1zbPml1hx09Ee96JcfwrpLq9HI"
        })
        .exec()
        .catch(e => {
            console.error(e)
            return {
                data: {
                    payload: []
                }
            }
        })

    return {
        props: {
            ssg: {
                introVideo: introVideo[0] || null
            }
        }
    }
}


export const Video = () => {
    const {introVideo} = useSSG();

}
